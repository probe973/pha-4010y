    Step 1: Find rate per minute using patient mass:

    ${{include.q.rate | round: 2 | replace: '.00', '' | replace: '.0', ''}} \text{mg/kg/min} \times {{include.q.mass | round: 2 | replace: '.00', '' | replace: '.0', ''}} \text{kg} = {{ratetime | round: 2 | replace: '.00', '' | replace: '.0', ''}} \text{mg/min}$

    ${{ratetime | round: 2 | replace: '.00', '' | replace: '.0', ''}} \text{mg/min} \times {{include.q.time | round: 2 | replace: '.00', '' | replace: '.0', ''}} \text{min} = {{need | round: 2 | replace: '.00', '' | replace: '.0', ''}} \text{mg}$

    $\frac{ {{need | round: 2 | replace: '.00', '' | replace: '.0', ''}} }{ {{have | round: 2 | replace: '.00', '' | replace: '.0', ''}} } \times 100 \text{ml} = {{volume | round: 2 | replace: '.00', '' | replace: '.0', ''}} \text{ml}$
